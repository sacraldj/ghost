    followers = stats.get('followers', 0)
    retweets = stats.get('retweets', 0)
    likes = stats.get('likes', 0)
    
    influence = (
        self.config["influence"]["tweet"]["weight_followers"] * log(followers) +
        self.config["influence"]["tweet"]["weight_retweets"] * log(retweets + 1) +
